```python
# src/<package_module>/config.py
"""Client configuration and factory for maf-packager LLM providers."""
from __future__ import annotations

import os
from dataclasses import dataclass, field
from typing import Any


@dataclass
class ClientConfig:
    """Resolved LLM client configuration from environment variables."""

    client_type: str  # "foundry" | "openai" | "azure_openai"
    model: str
    foundry_endpoint: str | None = None
    foundry_credential_type: str = "default"  # "default" | "cli" | "managed_identity"
    openai_api_key: str | None = None
    azure_openai_endpoint: str | None = None
    azure_openai_api_key: str | None = None
    azure_openai_api_version: str = "2024-02-01"


def get_client_config() -> ClientConfig:
    """Auto-detect LLM client type from environment variables.

    Priority order: Foundry > Azure OpenAI > OpenAI.

    Raises:
        RuntimeError: If no LLM provider environment variables are set.
    """
    foundry_endpoint = os.environ.get("FOUNDRY_PROJECT_ENDPOINT")
    azure_endpoint = os.environ.get("AZURE_OPENAI_ENDPOINT")
    openai_key = os.environ.get("OPENAI_API_KEY")

    if foundry_endpoint:
        return ClientConfig(
            client_type="foundry",
            model=os.environ.get("FOUNDRY_MODEL", "gpt-4o"),
            foundry_endpoint=foundry_endpoint,
            foundry_credential_type=os.environ.get("FOUNDRY_CREDENTIAL_TYPE", "default"),
        )
    elif azure_endpoint:
        return ClientConfig(
            client_type="azure_openai",
            model=os.environ.get("AZURE_OPENAI_MODEL", "gpt-4o"),
            azure_openai_endpoint=azure_endpoint,
            azure_openai_api_key=os.environ.get("AZURE_OPENAI_API_KEY"),
            azure_openai_api_version=os.environ.get("AZURE_OPENAI_API_VERSION", "2024-02-01"),
        )
    elif openai_key:
        return ClientConfig(
            client_type="openai",
            model=os.environ.get("OPENAI_CHAT_MODEL", "gpt-4o"),
            openai_api_key=openai_key,
        )
    else:
        raise RuntimeError(
            "No LLM client configured. Set FOUNDRY_PROJECT_ENDPOINT, "
            "AZURE_OPENAI_ENDPOINT, or OPENAI_API_KEY."
        )


def build_client(config: ClientConfig | None = None) -> Any:
    """Instantiate and return the correct LLM client.

    Uses lazy imports so that provider packages are not required at import time.

    Args:
        config: If None, calls get_client_config() to auto-detect from env vars.

    Returns:
        A configured chat client instance (FoundryChatClient, OpenAIChatCompletionClient,
        or OpenAIChatClient).

    Raises:
        RuntimeError: If no LLM provider is configured.
        ValueError: If client_type is unrecognised.
    """
    if config is None:
        config = get_client_config()

    if config.client_type == "foundry":
        from agent_framework.foundry import FoundryChatClient  # lazy import

        if config.foundry_credential_type == "cli":
            from azure.identity import AzureCliCredential
            credential = AzureCliCredential()
        elif config.foundry_credential_type == "managed_identity":
            from azure.identity import ManagedIdentityCredential
            credential = ManagedIdentityCredential()
        else:
            from azure.identity import DefaultAzureCredential
            credential = DefaultAzureCredential()

        return FoundryChatClient(
            project_endpoint=config.foundry_endpoint,
            model=config.model,
            credential=credential,
        )

    elif config.client_type == "azure_openai":
        from agent_framework.openai import OpenAIChatCompletionClient  # lazy import

        return OpenAIChatCompletionClient(
            api_key=config.azure_openai_api_key,
            azure_endpoint=config.azure_openai_endpoint,
            model=config.model,
            api_version=config.azure_openai_api_version,
        )

    elif config.client_type == "openai":
        from agent_framework.openai import OpenAIChatClient  # lazy import

        return OpenAIChatClient(
            api_key=config.openai_api_key,
            model=config.model,
        )

    else:
        raise ValueError(
            f"Unknown client_type {config.client_type!r}. "
            "Must be one of: 'foundry', 'azure_openai', 'openai'."
        )
```

### 2.1 `ToolRegistry` class

```python
class ToolRegistry:
    """Maps string tool names to callable Python functions.

    Use register() or register_all() to populate the registry before calling
    AgentFactory.create(). Unknown names are silently skipped with a warning.
    """

    def __init__(self, tools: dict[str, Callable[..., Any]] | None = None) -> None:
        self._registry: dict[str, Callable[..., Any]] = dict(tools or {})

    def register(self, name: str, fn: Callable[..., Any]) -> None:
        """Register a single named tool callable."""
        self._registry[name] = fn

    def register_all(self, tools: dict[str, Callable[..., Any]]) -> None:
        """Bulk-register a dict of name → callable mappings."""
        self._registry.update(tools)

    def get(self, name: str) -> Callable[..., Any] | None:
        """Return the callable for name, or None if not registered."""
        return self._registry.get(name)

    def resolve_tools(self, names: list[str]) -> list[Callable[..., Any]]:
        """Return callables for all known names; skip unknown names with a warning."""
        import warnings

        result: list[Callable[..., Any]] = []
        for name in names:
            fn = self._registry.get(name)
            if fn is not None:
                result.append(fn)
            else:
                warnings.warn(
                    f"Unknown tool {name!r} — not registered in ToolRegistry. Skipping.",
                    stacklevel=2,
                )
        return result
```

### 2.2 `AgentFactory` class

```python
class AgentFactory:
    """Instantiates MAF Agent objects from AgentConfig."""

    def __init__(
        self,
        client: Any,
        tool_registry: ToolRegistry | None = None,
    ) -> None:
        self.client = client
        self.tool_registry = tool_registry or ToolRegistry()

    def create(self, config: AgentConfig) -> Agent:
        """Create a single MAF Agent from an AgentConfig."""
        tools = self.tool_registry.resolve_tools(config.tools)
        return Agent(
            name=config.name,
            description=config.description,
            instructions=config.instructions,
            client=self.client,
            tools=tools,
        )

    def create_all(self, configs: list[AgentConfig]) -> list[Agent]:
        """Create MAF Agent objects for every AgentConfig in the list."""
        return [self.create(cfg) for cfg in configs]
```

## Code Requirements

- `from __future__ import annotations` on both files.
- Lazy imports inside `build_client()` for all provider-specific packages (`azure.identity`, `agent_framework.foundry`, `agent_framework.openai`). Do not import these at module level.
- All public methods/functions must have complete type annotations.
- `ToolRegistry.resolve_tools` must **never raise** — silently skip unknown tools with a `warnings.warn`.
- No global mutable state — `ToolRegistry._registry` is instance-level only.
- Module-level docstring on both files.
- Use `dict(tools or {})` in `ToolRegistry.__init__` to avoid sharing the default dict across instances.
- `AgentFactory.create_all` returns `list[Agent]`, not a generator.
- Do not import `AgentConfig` from `parser.py` — import it from `<package_module>.types`.
